import logger from '../utils/logger.js';

export class KitBrain {
    constructor(context) {
        this.context = context;
        this.updateInterval = 60000; // Atualiza a cada 1 minuto
        
        // Status iniciais (Serão carregados do DB no init)
        this.needs = {
            energy: 100,  // Sono
            hunger: 100,  // Fome
            mood: 100,    // Alegria/Tédio
            aura: 50      // O "Status" social dela com você
        };

        this.currentAction = "idle"; // sleeping, eating, working, judging
    }

    async init() {
        // Carrega o último estado salvo no SQLite
        const saved = await this.context.core.memory.getKitStatus();
        if (saved) {
            this.needs = saved.needs;
            this.currentAction = saved.currentAction;
        }
        
        this.startLoop();
        logger.info("Cérebro da Kit iniciado: O tempo está passando para ela.");
    }

    startLoop() {
        setInterval(async () => {
            this.processDecay();
            this.checkPriorities();
            await this.saveState();
        }, this.updateInterval);
    }

    processDecay() {
        // O tempo consome os recursos
        if (this.currentAction !== 'sleeping') this.needs.energy -= 0.5;
        if (this.currentAction !== 'eating') this.needs.hunger -= 0.8;
        
        // Se ela não está interagindo com nada, o tédio sobe (mood desce)
        if (this.currentAction === 'idle') this.needs.mood -= 1;

        // Garante que não fique negativo
        Object.keys(this.needs).forEach(k => {
            this.needs[k] = Math.max(0, Math.min(100, this.needs[k]));
        });
    }

    checkPriorities() {
        // Lógica de decisão autônoma
        if (this.needs.hunger < 20 && this.currentAction !== 'eating') {
            this.currentAction = 'eating';
            logger.info("Kit decidiu que vai comer agora.");
        } else if (this.needs.energy < 15 && this.currentAction !== 'sleeping') {
            this.currentAction = 'sleeping';
            logger.info("Kit capotou de sono.");
        }
    }

    async saveState() {
        await this.context.core.memory.saveKitStatus(this.needs, this.currentAction);
    }

    // Método para o Painel de Config ou Unity forçar uma ação
    async forceAction(action, modifiers = {}) {
        this.currentAction = action;
        if (modifiers.energy) this.needs.energy += modifiers.energy;
        if (modifiers.hunger) this.needs.hunger += modifiers.hunger;
        await this.saveState();
    }
}
